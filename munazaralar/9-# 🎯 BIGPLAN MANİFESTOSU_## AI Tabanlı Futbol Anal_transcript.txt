### Graph-LSTM Encoder (Beta'nÄ±n Ã–nerisi)
```python
class GraphLSTMEncoder(nn.Module):
    def __init__(self, node_dim, hidden_dim, num_layers=2):
        super().__init__()
        self.gnn = GCNConv(node_dim, hidden_dim)
        self.lstm = nn.LSTM(hidden_dim, hidden_dim, num_layers, batch_first=True)
        self.attention_pool = GlobalAttention(gate_nn=nn.Linear(hidden_dim, 1))
    
    def forward(self, x, edge_index, batch):
        # 1. GNN ile spatial encoding
        x_graph = self.gnn(x, edge_index)
        
        # 2. Global Attention Pooling â†’ graf vektÃ¶rÃ¼
        pooled = self.attention_pool(x_graph, batch)  # (Batch, Embed)
        
        # 3. Sequence formatÄ±na Ã§evir (Batch, Seq, Embed)
        pooled_seq = pooled.view(batch_size, seq_len, -1)
        
        # 4. LSTM ile temporal encoding
        lstm_out, (h_n, c_n) = self.lstm(pooled_seq)
        
        return lstm_out, h_n
```

### LSTM-State-Space Core
```python
class LSTMStateSpaceCore(nn.Module):
    def __init__(self, input_dim, hidden_dim, state_dim):
        super().__init__()
        self.lstm = nn.LSTM(input_dim, hidden_dim, bidirectional=True)
        self.state_space = StateSpaceModel(input_dim, state_dim)
        
        # Bidirectional Cross-attention
        self.cross_attn_lstm_to_ss = nn.MultiheadAttention(embed_dim=hidden_dim * 2, num_heads=8)
        self.cross_attn_ss_to_lstm = nn.MultiheadAttention(embed_dim=state_dim, num_heads=8)
        
        self.fusion = nn.Linear(hidden_dim * 2 + state_dim, hidden_dim)
```

### TFT Decoder (Theta'nÄ±n Ã–nerisi)
```python
class TFTDecoder(nn.Module):
    def __init__(self, hidden_dim, num_static_vars, num_temporal_vars):
        super().__init__()
        # Variable Selection Network
        self.vsn_static = VariableSelectionNetwork(num_static_vars, hidden_dim)
        self.vsn_temporal = VariableSelectionNetwork(num_temporal_vars, hidden_dim)
        
        # Interpretable Multi-Head Attention
        self.attention = InterpretableMultiHeadAttention(hidden_dim, num_heads=4)
        
    def forward(self, lstm_output, static_features):
        # Variable selection (hangi feature Ã¶nemli?)
        selected_static, static_weights = self.vsn_static(static_features)
        
        # Attention (yorumlanabilirlik)
```

```python
class IntegratedModel(nn.Module):
    def __init__(self):
        super().__init__()
        self.graph_lstm = GraphLSTMEncoder()   # Encoder: spatial-temporal
        self.lstm_core = LSTMStateSpaceCore()   # Core dynamics
```

## 4.1 RDQL (Risk-constrained DQL)
**Karar:** Constrained MDP formÃ¼lasyonu ile risk limitleri
**Kaynak:** `Harika_bir_yol_haritasÄ±_Ã§Ä±kardÄ±k...` transcript

```python
class RDQL_Agent:
    def compute_loss(self, batch):
        # Standard Q-loss
        q_loss = F.mse_loss(q_values, target_q)

        # Cost constraint (risk limiti)
        constraint_violation = F.relu(cost_values.mean() - self.risk_threshold)

        # Lagrangian relaxation
        total_loss = q_loss + self.lambda_multiplier * constraint_violation

        # Dual ascent (lambda gÃ¼ncelleme)
        self.lambda_multiplier += 0.01 * constraint_violation.item()
```

### GeliÅŸmiÅŸ Strateji: CVaR-Constrained Thompson Sampling
**Neden Bu YaklaÅŸÄ±m?**
- **Kelly Criterion**: Optimal stake ancak risk gÃ¶rmezden gelinir
- **Thompson Sampling**: Adaptif ancak risk limitleri yok
- **CVaR-Constrained Thompson**: Ä°kisinin avantajlarÄ±nÄ± birleÅŸtirir

```python
from scipy.stats import beta as beta_dist

def constrained_thompson_sampling(priors, cvar_limit=0.05, bankroll=1000):
    """
    CVaR kÄ±sÄ±tlÄ± Thompson Sampling bandit
    - priors: Her arm iÃ§in (alpha, beta) tuple listesi 
    - cvar_limit: Minimum kabul edilebilir CVaR (0.05 = %5)
    - bankroll: Mevcut bÃ¼tÃ§e
    """
    # Beta daÄŸÄ±lÄ±mÄ±ndan Ã¶rnekle
    samples = [beta_dist.rvs(alpha, beta_param) for alpha, beta_param in priors]
    
    # CVaR filtresi: Sadece risk eÅŸiÄŸini aÅŸan aksiyonlar
    valid_actions = [
        i for i in range(len(samples)) 
        if np.percentile([samples[i]], 5) >= cvar_limit  # %5 VaR kontrolÃ¼
    ]
    
    if not valid_actions:
        return None, 0  # HiÃ§bir aksiyon uygun deÄŸil - bekle
    
    # En yÃ¼ksek expected value olan aksiyonu seÃ§
    best_action = max(valid_actions, key=lambda i: samples[i])
    
    # CVaR-constrained stake (risk limiti dikkate alÄ±narak)
    stake = min(
        bankroll * 0.05,  # Max %5 single bet
        bankroll * samples[best_action] * 0.3  # %30 fractional
    )
    
    return best_action, stake
```
